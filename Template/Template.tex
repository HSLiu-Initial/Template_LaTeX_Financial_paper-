    %\title{\zihao{3}\textbf{你的标题}}

                aftername={~}                          
                                                ]{section}
                                                
    \CTEXsetup[ name={（,）},
                number={\chinese{subsection}},
                format={\raggedright\bfseries\zihao{-4}}

                                                ]{subsection}
